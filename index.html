# Algoritmo de Resolução Gauss-Jordan desenvolvido manualmente
def resolver_gauss_jordan(A, b):
    n = len(b)
    for i in range(n):
        pivot = A[i][i]
        if pivot == 0:
            for k in range(i+1, n):
                if A[k][i] != 0:
                    A[i], A[k] = A[k], A[i]
                    b[i], b[k] = b[k], b[i]
                    pivot = A[i][i]
                    break
        if pivot == 0: continue
        for j in range(n):
            A[i][j] = A[i][j] / pivot
        b[i] = b[i] / pivot
        for k in range(n):
            if k != i:
                fator = A[k][i]
                for j in range(n):
                    A[k][j] = A[k][j] - fator * A[i][j]
                b[k] = b[k] - fator * b[i]

Exercício Extra: Roteamento pelo Vizinho Mais Próximo

Instrução:

Implementar a heurística do Vizinho Mais Próximo para otimização do caminho em coordenadas espaciais e cálculo associado de consumo energético.

import math

def distancia(p1, p2):
    return math.sqrt((p1[0]-p2[0])**2 + (p1[1]-p2[1])**2)

def vizinho_mais_proximo(pontos):
    nao_visitados = pontos[:]
    atual = (0, 0)
    rota = []
    total = 0
    while nao_visitados:
        proximo = min(nao_visitados, key=lambda p: distancia(atual, p))
        total += distancia(atual, proximo)
        rota.append(proximo)
        atual = proximo
        nao_visitados.remove(proximo)
    total += distancia(atual, (0,0))
    return rota, total
